001package sztejkat.abstractfmt.testsuite;
002import org.junit.runner.RunWith;
003import org.junit.runners.Suite;
004import org.junit.AfterClass;
005import org.junit.runners.Suite.SuiteClasses;
006/**
007        A demo test-suite showing how to construct test
008        suite made of set of test-cases.
009*/
010@RunWith(Suite.class)   //Tell that it is a composite test
011@SuiteClasses({DemoTestCase.class})     //list, using {x.class,y.class,....} test components.
012public class DemoTestSuite<TDUT extends StringBuilder> extends ATestCase<TDUT>
013{       
014        //With this method we do release any factory assigned to
015        //a field, so if something conflicts set-up of next test
016        //suite implementation may detect it.
017        @AfterClass public static void releaseTest()
018        {
019                System.out.println("Releasing test factory");
020                DemoTestSuite.FACTORY = null;
021        };
022};
