001package sztejkat.abstractfmt.testsuite;
002import java.io.IOException;
003/**
004        A base of test cases.
005        <p>
006        Each test case is supposed to run some sub-set of
007        tests on <code>TDUT</code> contract using
008        instances created with {@link #create} methods.
009        <p>
010        For how to implement inspect source of 
011        {@link DemoTestCase}.
012        <p>
013        For how to collate many test cases in 
014        test suite inspect source code of {@link DemoTestSuite}.
015        <p>
016        For how to actually run test suite on some implementation
017        of <code>TDUT</code> inspect source code of {@link TestDemoTestSuite_A}. 
018*/
019public abstract class ATestCase<TDUT extends Object>
020                                //Note: remove this test dependency later.
021                                extends sztejkat.utils.test.ATest
022{
023                        /** A field which must be initialized with a 
024                        specific test factory */
025                        public static IDeviceUnderTestFactory<?> FACTORY;  
026                
027                        
028        /** Uses {@link #FACTORY} to produce device under test
029        @return new device under test 
030        @throws IOException if factory have thrown.
031        */
032        @SuppressWarnings({"unchecked"})
033        protected TDUT create()throws IOException
034        {
035                assert(FACTORY!=null):"Static factory is not initialized";
036                return ((IDeviceUnderTestFactory<TDUT>)(FACTORY)).create(); 
037        };
038};
