001package sztejkat.abstractfmt.testsuite;
002import java.io.IOException;
003
004/**
005        A factory which is set to test case factory
006        field and is used to produce test cases.
007        <p>
008        Instances of this class must be state-less.
009*/
010public interface IDeviceUnderTestFactory<T extends Object>
011{
012        /** Creates test case.
013        @return new independent instance 
014        @throws IOException if failed.
015        */
016        public T create()throws IOException;
017};
