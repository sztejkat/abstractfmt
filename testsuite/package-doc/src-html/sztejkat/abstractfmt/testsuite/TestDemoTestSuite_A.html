001package sztejkat.abstractfmt.testsuite;
002import java.io.IOException;
003import org.junit.BeforeClass;
004/**
005        A demo test-suite showing how to use
006        a test suite {@link DemoTestSuite}
007        with a specific implementation of 
008        "device under test".
009*/
010public class TestDemoTestSuite_A extends DemoTestSuite<StringBuilder>
011{
012        //In theory the static class initializer:
013        // static{ ...}
014        // is run when any class data or code is used
015        //for a first time. The tricky part is, that JUnit default runner
016        //which is encountering this class sees @RunWith(Suite) and does
017        //NOT in fact initialize this class unless there is some code
018        //which it is bound to run. Thous instead of static initializer
019        //we are bound to use @org.junit.BeforeClass annotated method. 
020                
021        @BeforeClass public static void setupTestSuiteFactory()
022        {
023                System.out.println("TestDemoTestSuite_A.setupTestSuiteFactory()");
024                //Validate if some composite test already set it up?
025                assert(DemoTestSuite.FACTORY==null):"Conflicting tests initialization";
026                DemoTestSuite.FACTORY = new IDeviceUnderTestFactory<StringBuilder>()
027                {
028                        public StringBuilder create()throws IOException 
029                        {
030                                System.out.println("TestDemoTestSuite_A factory creates DUT");
031                                return new StringBuilder();
032                        };
033                };
034        };
035};
