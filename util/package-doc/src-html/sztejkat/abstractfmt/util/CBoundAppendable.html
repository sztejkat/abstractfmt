001package sztejkat.abstractfmt.util;
002import sztejkat.abstractfmt.EFormatBoundaryExceeded;
003import java.io.IOException;
004/**
005        A string-builder like class which implements {@link Appendable}
006        contract which throws {@link EFormatBoundaryExceeded}
007        if an attempt is made to put too many data into it.
008        <p>
009        This class may be used when You do not trust the char processing 
010        stream routines to trully obey string size limits.
011*/
012public class CBoundAppendable implements Appendable,CharSequence
013{
014                                /** Buffer */
015                                private final char [] buffer;
016                                /** Where to put char and length of buffer */
017                                private int at;
018                                /** Hash code 
019                                @see #hashCode
020                                */
021                                private int hash;
022                                
023        /* ********************************************************
024        
025                        Construction
026        
027        *********************************************************/
028        /** Creates, pre allocating buffer of specified limit
029        @param limit non-negative, can be zero.
030        */
031        public CBoundAppendable(int limit)
032        {
033                assert(limit>=0);
034                this.buffer = new char[limit];
035                this.at = 0;
036        };
037        /* ********************************************************
038        
039                        Access
040        
041        *********************************************************/
042        /** Capactiy of buffer 
043        @return capacity of buffer, in chars
044        */
045        public final int capacity(){ return buffer.length; };
046        /** Resets to zero size */
047        public final void reset(){ at = 0; hash =0;};
048        /** Transforms to independent string 
049        @return string representation of collected chars
050        */
051        public final String toString(){ return new String(buffer,0,at); };
052        /* ********************************************************
053        
054                        CharSequence    
055        
056        *********************************************************/
057        /** Returns length used in a buffer 
058        @return length of string in buffer */
059        public final int length(){return at; };
060        public final boolean isEmpty(){ return at==0;};
061        public final char charAt(int index)
062        {
063                if (index>=at) throw new IndexOutOfBoundsException(index+" is not in 0..."+at+"-1");
064                return buffer[index];
065        };
066        public final CharSequence subSequence(int start,int end)
067        {
068                return toString().subSequence(start,end);
069        };
070        /* ********************************************************
071        
072                        Appendable      
073        
074        *********************************************************/
075        private void appendImpl(CharSequence csq,
076                                       int start,
077                                       int end)throws EFormatBoundaryExceeded
078    {
079                for(int i=start; i<end; i++)
080                        {
081                                if (at>=buffer.length) throw new EFormatBoundaryExceeded("String capacity "+at+" is already used up. Some string in stream is too long");
082                                buffer[at]=csq.charAt(i);
083                                at++;
084                        };
085    };
086        public Appendable append(CharSequence csq)throws IOException
087        {
088                if (csq==null)
089                { 
090                        append('n'); append('u'); append('l'); append('l');
091                }else
092                {
093                        appendImpl(csq,0,csq.length());
094                };
095                return this;
096        };
097        public Appendable append(CharSequence csq,
098                                       int start,
099                                       int end)throws IOException
100    {
101        if (csq==null)
102                { 
103                        append('n'); append('u'); append('l'); append('l');
104                }
105                else
106                {
107                        appendImpl(csq,start,end);
108                }
109                return this;
110    };
111    public Appendable append(char c)throws IOException
112    {
113        if (at>=buffer.length) throw new EFormatBoundaryExceeded("String capacity "+at+" is already used up. The string \""+this.toString()+""+c+"\" is too long");
114                buffer[at]=c;
115                at++;
116                return this;
117    };
118    /** Copied from JDK8 String.hashCode
119    to match standard.
120    @return hashCode.
121    */
122    public final int hashCode()
123    {
124        /* Please notice the inherent inefficiency
125           The 31*h will cross the 32 bit boundary
126           after about 20 characters or so. If computation
127           will continue after next 20 characters the
128           effect of first characters will get lost.
129
130           Why they have choosen unlimited hash, when
131           hash is just used to guess if texts are
132           "similiar"?
133        */
134        int h = hash;
135        final int L = length();
136        if (h == 0 && L > 0) {
137            final char val[] = buffer;
138
139            for (int i = 0; i < L; i++) {
140                h = 31 * h + val[i];
141            }
142            hash = h;
143        }
144        return h;
145    }
146    /** A quick test, faster than 
147    <code>s.equals(this.toString)</code>
148    because it avoids creation of a temporary object
149    @param s string to compare with, non-null
150    @return true if content is equal
151    */
152    public boolean equalsString(String s)
153    {
154        assert(s!=null);
155        //Note: Objects with different hashes can't be 
156        //the same.
157        //Notice however, that computing hash code
158        //may be more time consuming than
159        //exact comparison for longer strings.
160        //But it may be more efficient if we
161        //compare the same string with multiple
162        //textes, since then hascode is cached.
163        //This is precisely why it could be used
164        //in switch statement.
165        //
166        //We can easliy check if we know own hascode,
167        //but we can't test if s knows it.
168        int L = length();
169        if (L!=s.length()) return false;
170        if ((L<=32)&&(hash!=0))   //optimizing guess.
171        {
172                if (hashCode()!=s.hashCode()) return false;
173        };
174        for(int i=L;--i>=0;)
175        {
176                if (buffer[i]!=s.charAt(i)) return false;
177        };
178        return true;
179    };
180};
