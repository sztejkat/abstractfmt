001package sztejkat.abstractfmt.util;
002/**
003        A simplified toolbox class which is intended to 
004        be used for testing purpose of character based formats
005        and provides an ability to write and read characters from
006        a certain buffer.
007        <p>
008        <i>Notes: This class could be a pair {@link java.io.PipedReader} and {@link java.io.PipedWriter}
009        if those could be tuned to have a different behavior on circular buffer overrun.
010        The primary intent of those pipes was to provide a blocking I/O to exchange
011        data between <u>threads</u>. In this package and for testing I need a 
012        different approach.
013        </i>    
014        <h1>Thread safety</h1>
015        This class is NOT thread safe and is intended to be used in single
016        thread to buffer and read characters, possibly interleaving those operations.
017*/
018public final class CCharExchangeBuffer
019{       
020                        /** How much of size will be added to {@link #buffer}
021                        when it is necessary */
022                        private static final int GROWTH_INCREMENT = 1024;
023                        /** How much of data needs to be read from {@link #buffer}
024                        to move data to beginning of it. */
025                        private static final int MOVE_TRIGGER = GROWTH_INCREMENT/2;
026                        /** How much of data must be free in buffer after 
027                        move trigger to shrink it.
028                        The shrunken size will be current size + {@link #GROWTH_INCREMENT}
029                        so this value must be larger than {@link #GROWTH_INCREMENT} */
030                        private static final int SHRINK_TRIGGER = GROWTH_INCREMENT*2; //<-- must be greater.
031                        /** A buffer, which may adaptively grow */
032                        private char [] buffer;
033                        /** Where to write next character */
034                        private int write_at;
035                        /** Where to read next character */
036                        private int read_at;
037                        
038                public CCharExchangeBuffer()
039                {
040                        this.buffer = new char[GROWTH_INCREMENT];                       
041                };
042                /** True if any character is available for reading
043                @return false if do not have anything to read */
044                private boolean available()
045                {
046                        return (read_at<write_at);
047                };
048                /** Number of characters available for reading 
049                @return chars */
050                private int length()
051                {
052                        return write_at-read_at;
053                };
054                
055                public String toString()
056                {
057                        return new String(
058                                                buffer,//char[] value, int offset, int count)
059                                                read_at,
060                                                length());
061                };
062                /** Reads character or returns -1
063                @return 0...0xFFFF or -1 if nothing to read
064                */
065                public int read()
066                {
067                        if (!available()) return -1;
068                        int at = read_at;
069                        char c = buffer[at++];
070                        if (at>MOVE_TRIGGER)
071                        {
072                                        //move data or shrink and move.
073                                        int wat = write_at;
074                                        int L = wat - at;
075                                        int free_after_move = buffer.length-L;
076                                        if (free_after_move>SHRINK_TRIGGER)
077                                        {
078                                                //shrink AND move
079                                                char [] nb = new char[L+GROWTH_INCREMENT];
080                                                System.arraycopy(buffer,at, nb, 0, L);
081                                                buffer = nb;
082                                        }else
083                                        {
084                                                //just move
085                                                System.arraycopy(buffer,at, buffer, 0, L);
086                                        };
087                                        //update pointers
088                                        read_at =0;
089                                        write_at=L;
090                        }else
091                        {
092                                //just update read pointer.
093                                read_at = at;
094                        };
095                        return c;
096                };
097                /** Writes character possibly enlarging buffer.
098                No buffer limiting is applied 
099                @param c character to write
100                */
101                public void write(char c)
102                {
103                        int at = write_at;
104                        //check if has free space?
105                        if (at==buffer.length)
106                        {
107                                buffer = java.util.Arrays.copyOf(buffer, at + GROWTH_INCREMENT);
108                        };
109                        //store
110                        buffer[at++] = c;
111                        //update pointer.
112                        write_at = at;
113                };
114                /** Writes text possibly enlarging buffer.
115                No buffer limiting is applied 
116                @param s text to write
117                */
118                public void write(CharSequence s)
119                {
120                        for(int i=0,n=s.length(); i<n; i++)
121                                                                        write(s.charAt(i));
122                };
123

126                /* ********************************************************************************
127                
128                
129                
130                                        Junit org test arena
131                
132                
133                        Note: tests might need update if GROWTH_INCREMENT/MOVE_TRIGGER/SHRINK_TRIGGER
134                        are changed.
135                
136                * *********************************************************************************/
137                public static final class Test extends sztejkat.utils.test.ATest
138                {
139                        private static char of(int i){ return (char)((i*31) +i ); }
140                        private void testReadAndWrite(final int chunk_size, CCharExchangeBuffer b)
141                        {
142                                for(int i=0;i<chunk_size; i++)
143                                {
144                                        b.write(of(i));
145                                };
146                                org.junit.Assert.assertTrue(b.length()==chunk_size);
147                                for(int i=0;i<chunk_size; i++)
148                                {
149                                        org.junit.Assert.assertTrue(b.available());
150                                        org.junit.Assert.assertTrue(b.read()==of(i));
151                                };
152                        };
153                        @org.junit.Test public void testReadAndWrite1()
154                        {       
155                                enter();
156                                /*
157                                                We do small, sub growth/move/shring read-write test.
158                                */
159                                CCharExchangeBuffer b = new CCharExchangeBuffer();
160                                for(int i=0;i<GROWTH_INCREMENT/16-1; i++)
161                                {
162                                        testReadAndWrite(16,b);
163                                };
164                                leave();
165                        };
166                        
167                        
168                        @org.junit.Test public void testReadAndWrite2()
169                        {       
170                                enter();
171                                /*
172                                                do large sub growth/move/shring read-write test
173                                                which should trigger multiple increments and shrinks
174                                */
175                                CCharExchangeBuffer b = new CCharExchangeBuffer();
176                                for(int i=0;i<GROWTH_INCREMENT*4-1; i++)
177                                {
178                                        testReadAndWrite(128,b);
179                                };
180                                //And now take a look at buffer length
181                                int L = b.buffer.length;
182                                System.out.println("Buffer length after operation is "+L+" pointers are r:"+b.read_at+" w:"+b.write_at);
183                                org.junit.Assert.assertTrue( L<= SHRINK_TRIGGER );
184                                leave();
185                        };
186                        
187                        @org.junit.Test public void testReadAndWrite3()
188                        {       
189                                enter();
190                                /*
191                                                do large sub growth/move/shring read-write test
192                                                which should trigger multiple increments and shrinks
193                                                with large operations.
194                                */
195                                CCharExchangeBuffer b = new CCharExchangeBuffer();
196                                for(int i=0;i<10; i++)
197                                {
198                                        testReadAndWrite(GROWTH_INCREMENT+5,b);
199                                };
200                                //And now take a look at buffer length
201                                int L = b.buffer.length;
202                                System.out.println("Buffer length after operation is "+L+" pointers are r:"+b.read_at+" w:"+b.write_at);
203                                org.junit.Assert.assertTrue( L<= SHRINK_TRIGGER );
204                                leave();
205                        };
206                };
207};
